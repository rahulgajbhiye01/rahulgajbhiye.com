## Code Examples with Different Themes

### Dark Theme Code Block

```javascript filename="darkTheme.js" showLineNumbers
// This code block uses the default dark theme
const themes = {
  dark: {
    background: "#1e1e1e",
    foreground: "#d4d4d4",
    accent: "#007acc",
  },
  light: {
    background: "#ffffff",
    foreground: "#333333",
    accent: "#0066cc",
  },
};

function applyTheme(themeName) {
  const theme = themes[themeName];
  document.documentElement.style.setProperty("--bg-color", theme.background);
  document.documentElement.style.setProperty("--text-color", theme.foreground);
  document.documentElement.style.setProperty("--accent-color", theme.accent);
}

// Apply dark theme by default
applyTheme("dark");
```

## Complex Code Example

```ts filename="blog-system.ts" showLineNumbers
import { MDXRemote } from "next-mdx-remote/rsc";
import { serialize } from "next-mdx-remote/serialize";
import matter from "gray-matter";

interface BlogPost {
  slug: string;
  frontmatter: {
    title: string;
    date: string;
    excerpt: string;
    tags: string[];
    published: boolean;
  };
  content: string;
  mdxSource: any;
}

class BlogSystem {
  private posts: Map<string, BlogPost> = new Map();

  async loadPost(filePath: string): Promise<BlogPost> {
    const fileContent = await fs.readFile(filePath, "utf-8");
    const { data, content } = matter(fileContent);

    const mdxSource = await serialize(content, {
      parseFrontmatter: true,
      mdxOptions: {
        remarkPlugins: [],
        rehypePlugins: [],
      },
    });

    const post: BlogPost = {
      slug: path.basename(filePath, ".mdx"),
      frontmatter: data as BlogPost["frontmatter"],
      content,
      mdxSource,
    };

    this.posts.set(post.slug, post);
    return post;
  }

  getPost(slug: string): BlogPost | undefined {
    return this.posts.get(slug);
  }

  getAllPosts(): BlogPost[] {
    return Array.from(this.posts.values())
      .filter((post) => post.frontmatter.published)
      .sort(
        (a, b) =>
          new Date(b.frontmatter.date).getTime() -
          new Date(a.frontmatter.date).getTime()
      );
  }

  searchPosts(query: string): BlogPost[] {
    return this.getAllPosts().filter(
      (post) =>
        post.frontmatter.title.toLowerCase().includes(query.toLowerCase()) ||
        post.frontmatter.excerpt.toLowerCase().includes(query.toLowerCase()) ||
        post.content.toLowerCase().includes(query.toLowerCase())
    );
  }
}

export { BlogSystem, type BlogPost };
```
